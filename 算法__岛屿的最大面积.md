### 岛屿最大面积

```javascript
/**
 * @param {number[][]} grid
 * @return {number}
 */
var maxAreaOfIsland = function (grid) {
  if (grid === [] || grid === [[]]) {
    return 0;
  }
  const max = 0;
  const row = grid.length - 1;
  const column = grid[0].length - 1;
  const compare = function (i, j, num) {
    if (i < 0 || j < 0 || j > column || j > row) {
      return;
    }
    if (grid[i][j] === 1) {
      grid[i][j] = 0;
      num++;
    }
    compare(i - 1, j, num);
    compare(i + 1, j, num);
    compare(i, j - 1, num);
    compare(i, j + 1, num);
    return num;
  };
  for (let i = 0; i <= row; i++) {
    for (let j = 0; j <= column; j++) {
      max = Math.max(compare(i, j, 0), max);
    }
  }
  return max;
};
```
